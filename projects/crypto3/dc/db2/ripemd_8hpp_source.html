41  struct ripemd_compressor<128> : public basic_ripemd_compressor<128> {
42  static void process_block(state_type &state, const block_type &block) {
43  // ripemd works on two 'lines' in parallel
44  // all variables worked on by line 1 are suffixed with 1
45  // all variables for line 2 with 2
46 
47  word_type A1 = state[0], B1 = state[1], C1 = state[2], D1 = state[3];
48  word_type A2 = state[0], B2 = state[1], C2 = state[2], D2 = state[3];
49 
50  // round 1
51 #pragma clang loop unroll(full)
52  for (int j = 0; j < 16; ++j) {
53  policy_type::transform<policy_type::f1>(A1, B1, C1, D1, block[policy_type::r1[j]], 0x00000000,
54  policy_type::s1[j]);
55  policy_type::transform<policy_type::f4>(A2, B2, C2, D2, block[policy_type::r2[j]], 0x50a28be6,
56  policy_type::s2[j]);
57  }
58  // round 2
59 #pragma clang loop unroll(full)
60  for (int j = 16; j < 32; ++j) {
61  policy_type::transform<policy_type::f2>(A1, B1, C1, D1, block[policy_type::r1[j]], 0x5a827999,
62  policy_type::s1[j]);
63  policy_type::transform<policy_type::f3>(A2, B2, C2, D2, block[policy_type::r2[j]], 0x5c4dd124,
64  policy_type::s2[j]);
65  }
66  // round 3
67 #pragma clang loop unroll(full)
68  for (int j = 32; j < 48; ++j) {
69  policy_type::transform<policy_type::f3>(A1, B1, C1, D1, block[policy_type::r1[j]], 0x6ed9eba1,
70  policy_type::s1[j]);
71  policy_type::transform<policy_type::f2>(A2, B2, C2, D2, block[policy_type::r2[j]], 0x6d703ef3,
72  policy_type::s2[j]);
73  }
74  // round 4
75 #pragma clang loop unroll(full)
76  for (int j = 48; j < 64; ++j) {
77  policy_type::transform<policy_type::f4>(A1, B1, C1, D1, block[policy_type::r1[j]], 0x8f1bbcdc,
78  policy_type::s1[j]);
79  policy_type::transform<policy_type::f1>(A2, B2, C2, D2, block[policy_type::r2[j]], 0x00000000,
80  policy_type::s2[j]);
81  }
82 
83  word_type T = state[1] + C1 + D2;
84  state[1] = state[2] + D1 + A2;
85  state[2] = state[3] + A1 + B2;
86  state[3] = state[0] + B1 + C2;
87  state[0] = T;
88  }
